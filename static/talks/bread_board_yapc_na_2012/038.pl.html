MongoDBx::Bread::Board::Container

container MyApp => as {
    container MongoDBx::Bread::Board::Container->new(
        name            => 'myapp_db',
        host            => 'localhost',
        database_layout => {
            user_db => ['standard_users', 'super_users'],
        },
    );

    service authenticator => (
        class => 'Authenticator',
        dependencies => ['myapp_db/user_db/standard_users'],
    );
};
